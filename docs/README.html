Quick Start

# Local development (Docker Compose)
cp .env.example .env       # fill in DB + JWT secrets
docker compose up -d       # starts postgres, backend, frontend

# Run tests
cd codebase/backend  && ./gradlew cleanTest test --no-daemon
cd codebase/frontend && npx vitest run

# Default ports: frontend=3000, backend=8080, postgres=5432
